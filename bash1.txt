# 1. Открыть домашнюю директорию
cd ~
echo "cd ~" >> bash1.txt

# 2. Определить имя текущей папки
pwd
echo "pwd" >> bash1.txt

# 3. Создать каталог с именем test1
mkdir test1
echo "mkdir test1" >> bash1.txt

# 4. Перейти в папку test1
cd test1
echo "cd test1" >> bash1.txt

# 5. Создать файлы 1, 2 и 3
touch 1 2 3
echo "touch 1 2 3" >> bash1.txt

# 6. Проверить содержимое каталога test1
ls
echo "ls" >> bash1.txt

# 7. Перейти в домашнюю директорию
cd ~
echo "cd ~" >> bash1.txt

# 8. Создать папку test2
mkdir test2
echo "mkdir test2" >> bash1.txt

# 9. Удалить папку test2
rmdir test2
echo "rmdir test2" >> bash1.txt

# 10. Удалить файл 2 из папки test1
rm ~/test1/2
echo "rm ~/test1/2" >> bash1.txt

# 11. Создать папку test3 и добавить в нее два файла
mkdir test3
touch test3/file1 test3/file2
echo "mkdir test3" >> bash1.txt
echo "touch test3/file1 test3/file2" >> bash1.txt

# 12. Удалить папку test3
rm -r test3
echo "rm -r test3" >> bash1.txt

# 13. Создать папку test4
mkdir test4
echo "mkdir test4" >> bash1.txt

# 14. Переместить файлы 1 и 3 из папки test1 в папку test4
mv ~/test1/1 ~/test1/3 ~/test4
echo "mv ~/test1/1 ~/test1/3 ~/test4" >> bash1.txt

# 15. Добавить в файл 1 три строки со словами line
echo -e "line\nline\nline" >> ~/test4/1
echo "echo -e 'line\nline\nline' >> ~/test4/1" >> bash1.txt

# 16. Посмотреть содержимое файла 1
cat ~/test4/1
echo "cat ~/test4/1" >> bash1.txt

# 17. Добавить в файл 3 три строки со словами line
echo -e "line\nline\nline" >> ~/test4/3
echo "echo -e 'line\nline\nline' >> ~/test4/3" >> bash1.txt

# 18. Просмотреть содержимое двух файлов (1 и 3) сразу
cat ~/test4/1 ~/test4/3
echo "cat ~/test4/1 ~/test4/3" >> bash1.txt

# 19. Используя редактор, заменить все строки в файле 1
# Откройте файл 1 в любом редакторе, например, nano или vim
